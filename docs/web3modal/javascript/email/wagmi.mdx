import TabItem from '@theme/TabItem'

<Tabs>
<TabItem value="default" label="Default">

If you're using the `defaultWagmiConfig` function you can enable email login by just adding `enableEmail: true` as a configuration parameter.

```ts
import { createWeb3Modal, defaultWagmiConfig } from '@web3modal/wagmi'

//...

const wagmiConfig = defaultWagmiConfig({
  chains,
  projectId,
  metadata,
  /* highlight-add-start */
  enableEmail: true // Optional - false by default
  /* highlight-add-end */
})
```

</TabItem>
<TabItem value="custom" label="Custom">

If you're using the `createConfig` function from Wagmi, you can enable email login by importing and adding the `emailConnector` function in your configuration.

```ts
import {
  createWeb3Modal
  /* highlight-add-start */
  emailConnector
  /* highlight-add-end */
} from '@web3modal/wagmi'

//...

const wagmiConfig = createConfig({
    //...
  connectors:[
    //...
    /* highlight-add-start */
    emailConnector({ chains, options: { projectId } })
    /* highlight-add-end */
  ]
})
```

</TabItem>
