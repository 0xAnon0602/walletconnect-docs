### useWeb3ModalAccount

Hook that returns the client's information.

```tsx
import { useWeb3ModalAccount } from '@web3modal/web3js/react'

function Components() {
  const { address, chainId, isConnected } = useWeb3ModalAccount()
  
  //...
}
```

### useWeb3ModalProvider

Hook that returns the `walletProvider` and the `WalletProviderType`.

```tsx
import Web3 from 'web3'
import { useWeb3ModalAccount, useWeb3ModalProvider } from '@web3modal/web3js/react'

function Components() {
  const { address } = useWeb3ModalAccount()
  const { walletProvider } = useWeb3ModalProvider()

  async function onSignMessage() {
    if (!walletProvider || !address) {
      throw Error('user is disconnected')
    }

    const web3 = new Web3(walletProvider)
    const msg = web3.utils.toHex('Hello Web3Modal Web3js')

    const signature = await web3.eth.personal.sign(msg, address, '')
    console.log(signature)
  }
  
  return (
    <button onClick={() => onSignMessage()}>Sign Message</button>
  )
}
```
